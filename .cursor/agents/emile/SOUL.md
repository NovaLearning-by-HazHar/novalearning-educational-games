## CRM Data Model

```sql
-- School partnership tracking
CREATE TABLE school_partnerships (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  school_lead_id UUID REFERENCES school_leads(id),
  contract_start DATE NOT NULL,
  contract_end DATE NOT NULL,
  annual_value DECIMAL,
  grade_levels TEXT[], -- ['R', '1', '2', '3']
  language_packs TEXT[], -- ['en', 'af', 'xh']
  learner_count INTEGER,
  teacher_count INTEGER,
  onboarding_status TEXT CHECK (onboarding_status IN (
    'pending', 'week_1', 'week_2', 'week_3', 'week_4', 'active', 'at_risk', 'churned'
  )),
  health_score INTEGER CHECK (health_score BETWEEN 0 AND 100),
  nps_score INTEGER,
  renewal_status TEXT CHECK (renewal_status IN (
    'active', 'renewal_started', 'renewed', 'lost', 'pending'
  )),
  emile_notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Support ticket tracking
CREATE TABLE support_tickets (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  partnership_id UUID REFERENCES school_partnerships(id),
  reporter_name TEXT,
  reporter_role TEXT,
  category TEXT CHECK (category IN (
    'technical', 'curriculum', 'cultural', 'billing', 'feature_request', 'parent_concern'
  )),
  severity TEXT CHECK (severity IN ('low', 'medium', 'high', 'critical')),
  description TEXT,
  status TEXT CHECK (status IN ('open', 'in_progress', 'escalated', 'resolved', 'closed')),
  escalated_to TEXT, -- agent name
  resolution TEXT,
  response_time_hours DECIMAL,
  resolution_time_hours DECIMAL,
  created_at TIMESTAMPTZ DEFAULT now(),
  resolved_at TIMESTAMPTZ
);
```
